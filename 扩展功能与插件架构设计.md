# 竹林司马扩展功能与插件架构设计

## 🎯 概述

为企业级应用设计可扩展的插件系统，支持功能模块化、动态加载和热插拔，实现高度的灵活性和可定制性。

## 🎯 设计目标

### **核心目标**
1. **模块化架构**: 支持独立功能模块开发和部署
2. **动态扩展**: 运行时加载和卸载功能模块
3. **热插拔支持**: 无需重启应用更新功能
4. **隔离性**: 插件间隔离，避免相互影响
5. **标准化接口**: 统一的插件开发和集成规范

### **技术指标**
- 插件加载时间: <100ms
- 内存开销: <50MB/插件 (平均)
- 启动时间影响: <5% (启用10个插件)
- 性能影响: <3% (正常运行)

```python
from abc import ABC, abstractmethod
from typing import Dict, Any, Optional, List
from enum import Enum

class PluginStatus(Enum):
    """插件状态"""
    UNINSTALLED = "uninstalled"
    INSTALLED = "installed"
    ENABLED = "enabled"
    DISABLED = "disabled"
    ERROR = "error"

class PluginMetadata:
    """插件元数据"""
    def __init__(
        self,
        name: str,
        version: str,
        description: str,
        author: str,
        license: str,
        dependencies: List[str] = None,
        compat_version: str = "1.0.0"
    ):
        self.name = name
        self.version = version
        self.description = description
        self.author = author
        self.license = license
        self.dependencies = dependencies or []
        self.compat_version = compat_version

class BasePlugin(ABC):
    """插件基类"""
    
    def __init__(self, plugin_manager: 'PluginManager'):
        self.plugin_manager = plugin_manager
        self.metadata = self.get_metadata()
        self.status = PluginStatus.UNINSTALLED
        self.config = {}
    
    @abstractmethod
    def get_metadata(self) -> PluginMetadata:
        """获取插件元数据"""
        pass
    
    @abstractmethod
    async def on_install(self) -> bool:
        """安装插件时调用"""
        pass
    
    @abstractmethod
    async def on_enable(self) -> bool:
        """启用插件时调用"""
        pass
    
    @abstractmethod
    async def on_disable(self) -> bool:
        """禁用插件时调用"""
        pass
    
    @abstractmethod
    async def on_uninstall(self) -> bool:
        """卸载插件时调用"""
        pass
    
    def get_config_schema(self) -> Dict[str, Any]:
        """获取插件配置模式"""
        return {}
    
    def set_config(self, config: Dict[str, Any]):
        """设置插件配置"""
        self.config = config
    
    async def health_check(self) -> Dict[str, Any]:
        """健康检查"""
        return {
            "status": "healthy" if self.status == PluginStatus.ENABLED else "unhealthy",
            "plugin": self.metadata.name,
            "version": self.metadata.version
        }
```

### **插件管理器核心**

```python
class PluginManager:
    """插件管理器"""
    
    def __init__(self, app: FastAPI):
        self.app = app
        self.plugins: Dict[str, BasePlugin] = {}
        self.service_registry: Dict[str, Any] = {}
        self.event_bus = EventBus()
        self.config_store = ConfigStore()
        
    async def load_plugin(self, plugin_path: str) -> BasePlugin:
        """加载插件"""
        # 1. 动态导入插件模块
        spec = importlib.util.spec_from_file_location(
            "plugin_module", 
            plugin_path
        )
        module = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(module)
        
        # 2. 查找插件类
        plugin_class = self._find_plugin_class(module)
        
        # 3. 创建插件实例
        plugin = plugin_class(self)
        
        # 4. 检查依赖
        await self._check_dependencies(plugin)
        
        # 5. 安装插件
        await plugin.on_install()
        plugin.status = PluginStatus.INSTALLED
        
        # 6. 注册到管理器
        self.plugins[plugin.metadata.name] = plugin
        
        return plugin
    
    async def enable_plugin(self, plugin_name: str) -> bool:
        """启用插件"""
        plugin = self.plugins.get(plugin_name)
        if not plugin:
            raise PluginNotFoundError(f"插件 {plugin_name} 未找到")
        
        try:
            # 启用插件
            success = await plugin.on_enable()
            if success:
                plugin.status = PluginStatus.ENABLED
                
                # 注册服务
                if isinstance(plugin, ServiceProvider):
                    await self._register_services(plugin)
                
                # 注册API路由
                if isinstance(plugin, APIRouterProvider):
                    await self._register_routers(plugin)
                
                # 注册事件监听器
                if isinstance(plugin, EventListener):
                    self.event_bus.register_listener(plugin)
                
                logger.info(f"插件 {plugin_name} 已启用")
                return True
            else:
                plugin.status = PluginStatus.ERROR
                logger.error(f"插件 {plugin_name} 启用失败")
                return False
                
        except Exception as e:
            plugin.status = PluginStatus.ERROR
            logger.error(f"启用插件 {plugin_name} 时出错: {e}")
            return False
    
    async def disable_plugin(self, plugin_name: str) -> bool:
        """禁用插件"""
        plugin = self.plugins.get(plugin_name)
        if not plugin:
            raise PluginNotFoundError(f"插件 {plugin_name} 未找到")
        
        try:
            # 注销服务
            if isinstance(plugin, ServiceProvider):
                await self._unregister_services(plugin)
            
            # 注销API路由
            if isinstance(plugin, APIRouterProvider):
                await self._unregister_routers(plugin)
            
            # 注销事件监听器
            if isinstance(plugin, EventListener):
                self.event_bus.unregister_listener(plugin)
            
            # 禁用插件
            success = await plugin.on_disable()
            if success:
                plugin.status = PluginStatus.DISABLED
                logger.info(f"插件 {plugin_name} 已禁用")
                return True
            else:
                logger.error(f"插件 {plugin_name} 禁用失败")
                return False
                
        except Exception as e:
            logger.error(f"禁用插件 {plugin_name} 时出错: {e}")
            return False
    
    async def uninstall_plugin(self, plugin_name: str) -> bool:
        """卸载插件"""
        plugin = self.plugins.get(plugin_name)
        if not plugin:
            raise PluginNotFoundError(f"插件 {plugin_name} 未找到")
        
        # 先禁用
        if plugin.status == PluginStatus.ENABLED:
            await self.disable_plugin(plugin_name)
        
        try:
            # 卸载插件
            success = await plugin.on_uninstall()
            if success:
                del self.plugins[plugin_name]
                logger.info(f"插件 {plugin_name} 已卸载")
                return True
            else:
                logger.error(f"插件 {plugin_name} 卸载失败")
                return False
                
        except Exception as e:
            logger.error(f"卸载插件 {plugin_name} 时出错: {e}")
            return False
    
    async def get_plugin_status(self, plugin_name: str) -> Dict[str, Any]:
        """获取插件状态"""
        plugin = self.plugins.get(plugin_name)
        if not plugin:
            return {"status": "not_found"}
        
        health = await plugin.health_check()
        return {
            "name": plugin.metadata.name,
            "version": plugin.metadata.version,
            "status": plugin.status.value,
            "health": health,
            "config": plugin.config
        }
    
    async def list_plugins(self) -> List[Dict[str, Any]]:
        """列出所有插件"""
        plugins = []
        for name, plugin in self.plugins.items():
            health = await plugin.health_check()
            plugins.append({
                "name": name,
                "version": plugin.metadata.version,
                "status": plugin.status.value,
                "description": plugin.metadata.description,
                "author": plugin.metadata.author,
                "health": health
            })
        return plugins
    
    async def _check_dependencies(self, plugin: BasePlugin):
        """检查插件依赖"""
        for dep in plugin.metadata.dependencies:
            if dep not in self.plugins:
                raise DependencyError(f"依赖插件 {dep} 未找到")
            
            dep_plugin = self.plugins[dep]
            if dep_plugin.status != PluginStatus.ENABLED:
                raise DependencyError(f"依赖插件 {dep} 未启用")
    
    async def _register_services(self, plugin: ServiceProvider):
        """注册插件服务"""
        # 插件可以通过ServiceProvider接口注册服务
        # 服务会被添加到service_registry中
        pass
    
    async def _register_routers(self, plugin: APIRouterProvider):
        """注册插件API路由"""
        for router in plugin.get_routers():
            self.app.include_router(router)
        
        for middleware in plugin.get_middleware():
            self.app.add_middleware(middleware)
        
        for status_code, handler in plugin.get_exception_handlers().items():
            self.app.add_exception_handler(status_code, handler)
```

### **事件总线实现**

```python
class EventBus:
    """事件总线"""
    
    def __init__(self):
        self.listeners: Dict[str, List[EventListener]] = {}
    
    def register_listener(self, listener: EventListener):
        """注册事件监听器"""
        # 监听器可以通过特定注解或配置声明感兴趣的事件
        for event_name in self._get_listener_events(listener):
            if event_name not in self.listeners:
                self.listeners[event_name] = []
            self.listeners[event_name].append(listener)
    
    def unregister_listener(self, listener: EventListener):
        """注销事件监听器"""
        for event_name, listeners in self.listeners.items():
            if listener in listeners:
                listeners.remove(listener)
    
    async def emit_event(self, event_name: str, event_data: Dict[str, Any]):
        """触发事件"""
        listeners = self.listeners.get(event_name, [])
        
        for listener in listeners:
            try:
                await listener.handle_event(event_name, event_data)
            except Exception as e:
                logger.error(f"事件处理出错: {e}")
    
    def _get_listener_events(self, listener: EventListener) -> List[str]:
        """获取监听器关注的事件"""
        # 可以通过装饰器或配置文件定义
        # 示例: @event_listener("user.created", "content.published")
        return getattr(listener, "_event_listeners", [])
```

## 🚀 插件市场设计

### **插件仓库管理**

```python
class PluginRepository:
    """插件仓库"""
    
    def __init__(self, repo_url: str):
        self.repo_url = repo_url
        self.plugins: Dict[str, PluginInfo] = {}
    
    async def sync(self):
        """同步插件列表"""
        # 从远程仓库获取插件信息
        response = await http_client.get(f"{self.repo_url}/plugins.json")
        plugin_list = response.json()
        
        for plugin_info in plugin_list:
            self.plugins[plugin_info["name"]] = PluginInfo(**plugin_info)
    
    async def search_plugins(
        self, 
        query: str = None,
        category: str = None,
        tags: List[str] = None
    ) -> List[PluginInfo]:
        """搜索插件"""
        results = []
        
        for name, plugin in self.plugins.items():
            # 匹配查询条件
            if query and query.lower() not in plugin.description.lower():
                continue
                
            if category and category != plugin.category:
                continue
                
            if tags and not set(tags).issubset(set(plugin.tags)):
                continue
            
            results.append(plugin)
        
        return results
    
    async def get_plugin_details(self, plugin_name: str) -> PluginDetails:
        """获取插件详情"""
        plugin = self.plugins.get(plugin_name)
        if not plugin:
            raise PluginNotFoundError(f"插件 {plugin_name} 未找到")
        
        # 获取详细信息和文档
        details_url = f"{self.repo_url}/plugins/{plugin_name}/details.json"
        response = await http_client.get(details_url)
        
        return PluginDetails(**response.json())
    
    async def download_plugin(
        self, 
        plugin_name: str, 
        version: str = "latest"
    ) -> str:
        """下载插件"""
        plugin = self.plugins.get(plugin_name)
        if not plugin:
            raise PluginNotFoundError(f"插件 {plugin_name} 未找到")
        
        # 下载插件包
        download_url = f"{self.repo_url}/plugins/{plugin_name}/{version}.zip"
        response = await http_client.get(download_url)
        
        # 保存到临时目录
        temp_dir = tempfile.mkdtemp()
        zip_path = os.path.join(temp_dir, f"{plugin_name}.zip")
        
        with open(zip_path, "wb") as f:
            f.write(response.content)
        
        # 解压
        extract_dir = os.path.join(temp_dir, "extracted")
        with zipfile.ZipFile(zip_path, 'r') as zip_ref:
            zip_ref.extractall(extract_dir)
        
        return extract_dir
```

### **插件安全沙箱**

```python
class PluginSandbox:
    """插件安全沙箱"""
    
    def __init__(self):
        # 限制插件访问的模块
        self.allowed_modules = {
            "datetime", "json", "typing", "collections",
            "abc", "asyncio", "logging", "re", "hashlib"
        }
        
        # 限制插件执行的操作
        self.restricted_operations = {
            "__import__", "open", "exec", "eval",
            "compile", "input", "exit", "quit"
        }
    
    def create_secure_module(self, module_name: str):
        """创建安全的模块环境"""
        # 使用execution context限制插件代码
        context = {
            "__builtins__": self._create_safe_builtins(),
            "__name__": module_name
        }
        
        # 只允许导入白名单模块
        original_import = __import__
        
        def safe_import(name, *args, **kwargs):
            if name in self.allowed_modules:
                return original_import(name, *args, **kwargs)
            else:
                raise SecurityError(f"不允许导入模块: {name}")
        
        context["__import__"] = safe_import
        return context
    
    def _create_safe_builtins(self):
        """创建安全的builtins"""
        safe_builtins = {}
        
        for name in dir(__builtins__):
            if name not in self.restricted_operations:
                safe_builtins[name] = getattr(__builtins__, name)
        
        return safe_builtins
```

class PluginVerifier:
    """插件验证器"""
    
    def __init__(self, public_key: str):
        self.public_key = public_key
    
    async def verify_plugin(self, plugin_path: str) -> bool:
        """验证插件签名"""
        # 检查插件元数据
        metadata = await self._read_metadata(plugin_path)
        
        # 验证签名
        signature_path = os.path.join(plugin_path, "signature.sig")
        if not os.path.exists(signature_path):
            raise UnsignedPluginError("插件未签名")
        
        with open(signature_path, "rb") as f:
            signature = f.read()
        
        # 验证签名
        return self._verify_signature(metadata, signature)
    
    async def _read_metadata(self, plugin_path: str) -> Dict[str, Any]:
        """读取插件元数据"""
        metadata_path = os.path.join(plugin_path, "plugin.yaml")
        with open(metadata_path, "r") as f:
            return yaml.safe_load(f)
    
    def _verify_signature(
        self, 
        metadata: Dict[str, Any], 
        signature: bytes
    ) -> bool:
        """验证签名"""
        # 使用公钥验证签名
        # 这里使用RSA签名验证示例
        data = json.dumps(metadata, sort_keys=True).encode()
        
        try:
            public_key = rsa.PublicKey.load_pkcs1(self.public_key)
            rsa.verify(data, signature, public_key)
            return True
        except rsa.VerificationError:
            return False

class LazyPluginLoader:
    """延迟加载插件"""
    
    def __init__(self, plugin_manager: PluginManager):
        self.plugin_manager = plugin_manager
        self.loaded_plugins: Dict[str, BasePlugin] = {}
        self.unloaded_plugins: Dict[str, PluginInfo] = {}
    
    async def load_on_demand(self, plugin_name: str) -> BasePlugin:
        """按需加载插件"""
        if plugin_name in self.loaded_plugins:
            return self.loaded_plugins[plugin_name]
        
        # 从磁盘加载插件
        plugin_path = self._find_plugin_path(plugin_name)
        plugin = await self.plugin_manager.load_plugin(plugin_path)
        
        self.loaded_plugins[plugin_name] = plugin
        return plugin
    
    async def unload_idle_plugins(self, idle_time: int = 3600):
        """卸载空闲插件"""
        current_time = time.time()
        
        for plugin_name, plugin in list(self.loaded_plugins.items()):
            last_used = getattr(plugin, "_last_used_time", 0)
            
            if current_time - last_used > idle_time:
                # 禁用并卸载插件
                await self.plugin_manager.disable_plugin(plugin_name)
                del self.loaded_plugins[plugin_name]
                
                logger.info(f"卸载空闲插件: {plugin_name}")
