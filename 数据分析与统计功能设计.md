### **聚合视图**

```sql
-- 用户活跃度日视图
CREATE VIEW user_daily_activity AS
SELECT 
    date,
    COUNT(DISTINCT user_id) as active_users,
    SUM(session_count) as total_sessions,
    AVG(total_duration) as avg_session_duration,
    SUM(page_views) as total_page_views
FROM user_behavior_stats
GROUP BY date;

-- 内容热度周视图
CREATE VIEW content_weekly_popularity AS
SELECT 
    EXTRACT(WEEK FROM date) as week_number,
    content_id,
    SUM(views) as total_views,
    SUM(likes) as total_likes,
    SUM(comments) as total_comments,
    RANK() OVER (PARTITION BY week_number ORDER BY SUM(views) DESC) as rank
FROM content_performance_stats
GROUP BY week_number, content_id;
```

## 🔄 数据处理流程

### **实时数据处理**

```python
class RealtimeDataProcessor:
    async def process_user_event(self, event: UserEvent):
        """处理用户事件"""
        # 1. 写入事件队列
        await self.event_queue.put(event)
        
        # 2. 更新实时计数器
        await self.update_realtime_counters(event)
        
        # 3. 触发实时分析
        await self.trigger_realtime_analysis(event)
    
    async def update_realtime_counters(self, event: UserEvent):
        """更新实时计数器"""
        key = f"realtime:{event.user_id}:{event.type}"
        await self.redis.incr(key)
        await self.redis.expire(key, 3600)  # 1小时过期
```

        self.redis = RedisCache()
        
    async def get_cached_analytics(self, key: str, ttl: int = 300):
        """获取缓存的分析数据"""
        cached = await self.redis.get(key)
        if cached:
            return json.loads(cached)

### **数据访问控制**
```python
class AnalyticsAccessControl:
    async def check_analytics_permission(
        self,
        user: User,
        analytics_type: str
    ) -> bool:
        """检查分析数据访问权限"""
        if user.is_admin:
            return True
            
        if analytics_type == "user_private":
            return user.has_permission("analytics.user.private")
            
        if analytics_type == "content_performance":
            return user.has_permission("analytics.content")
            
        return False
```
